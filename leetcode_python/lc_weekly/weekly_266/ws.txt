# brute force (O(N^2))

class Solution(object):
    def countVowelSubstrings(self, word):
        """
        :type word: str
        :rtype: int
        """
        n = len(word)
        total_count = 0
        vowels = set("aeiou")
        
        # Step 1: Pick the starting index 'i' of the substring
        for i in range(n):
            # A set to keep track of unique vowels found in the current substring
            seen_vowels = set()
            
            # Step 2: Expand the substring forward using ending index 'j'
            for j in range(i, n):
                char = word[j]
                
                # If we encounter a consonant, the substring becomes invalid.
                # Any further expansion from this 'i' is ruined, so we break early!
                if char not in vowels:
                    break
                    
                # Otherwise, it's a valid vowel. Add it to our unique tracking set.
                seen_vowels.add(char)
                
                # If our current substring contains exactly all 5 unique vowels, 
                # we increment our valid substring counter.
                if len(seen_vowels) == 5:
                    total_count += 1
                    
        return total_count

# slide window



class Solution(object):
    def countVowelSubstrings(self, word):
        """
        :type word: str
        :rtype: int
        """
        # Helper function: Counts all substrings consisting ONLY of vowels 
        # that contain AT MOST 'max_unique' distinct vowel characters.
        def countAtMost(max_unique):
            vowels = set("aeiou")
            cnt_map = {}
            l = 0
            ans = 0
            
            for r in range(len(word)):
                # CRITICAL STEP: If we hit a consonant, the window is instantly ruined.
                # Clear the tracking map and shift the left pointer past the consonant.
                if word[r] not in vowels:
                    cnt_map.clear()
                    l = r + 1
                    continue
                
                # Add the valid vowel to our sliding window frequency map
                cnt_map[word[r]] = cnt_map.get(word[r], 0) + 1
                
                # If we exceed our allowed unique vowel limit, shrink window from left
                while len(cnt_map) > max_unique:
                    cnt_map[word[l]] -= 1
                    if cnt_map[word[l]] == 0:
                        del cnt_map[word[l]]
                    l += 1
                
                # MATHEMATICAL INSIGHT: The number of valid substrings ending at index 'r'
                # is exactly equal to the length of the current window: (r - l + 1)
                ans += (r - l + 1)
                
            return ans

        # Substrings with EXACTLY 5 unique vowels = (At Most 5) - (At Most 4)
        return countAtMost(5) - countAtMost(4)
